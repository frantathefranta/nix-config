** Setting up backblaze B2 bucket
#+BEGIN_SRC
backblaze-b2 bucket create <bucketName> allPrivate \
   --default-server-side-encryption=SSE-B2 \
   --lifecycle-rules='[{"daysFromHidingToDeleting": 30, "daysFromUploadingToHiding": null, "fileNamePrefix": ""}]'
#+END_SRC
